#TIMEHORIZON
T = 365 * 24; // number of hours in two years


#NODE BATTERY
//TODO Aging, andere Investmentkosten bei steigender Kapazität
//TODO efficiency. Wie hoch ist die?
#PARAMETERS
    capex = 350; //kosten für 1kWp ca. 1300€
    depreciation = 12; //Abschreibungsdauer beträgt 20 Jahre
    annual_depreciation = capex / depreciation;
    efficiency = 0.85;
    power = 5; //maximale Leistung beträgt 3kw (d.h. es können max 3kwh pro Stunde in den Speicher geladen/entladen werden
    //self_consumption = 
#VARIABLES
    internal integer: capacity;
    internal: investment_cost;
    internal: soc[T];
    external: charge[T];
    external: discharge[T];
#CONSTRAINTS
    capacity >= 0;
    capacity <= 100; //Speicher darf max 100 kWh groß sein
    soc[t] >= 0;
    charge[t] >= 0;
    charge[t] <= power;
    discharge[t] >= 0;
    discharge[t] <= power;
    soc[t] <= capacity;
    charge[t] <= capacity - soc[t];
    discharge[t] <= soc[t];
    soc[t+1] == soc[t] + efficiency * charge[t] - discharge[t] / efficiency;
    soc[0] == 0;
    soc[0] == soc[T-1]; //batterie soll gleichen Ladestand am Anfang und am Ende haben
    investment_cost == annual_depreciation * capacity;
#OBJECTIVES
    min: investment_cost;
